tem_ElemSize

public pure function tem_ElemSize(tree, treeID) result(dx)

  pure function tem_ElemSize( tree, treeID ) result(dx)
    ! -------------------------------------------------------------------- !

    !> input elements
    integer(kind=long_k), intent(in) :: treeID
    !> size of element
    real(kind=rk) :: dx
